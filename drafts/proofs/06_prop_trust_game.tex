\begin{proof}[Proof of Proposition~\ref{prop:trust}]
The proposition requires (A2') attribution monotonicity and (G) guilt dominance ($G \equiv \gamma_H \lambda_H^{GUILT} > 1$).

%------------------------------------------------------------------------------
% PART (i)
%------------------------------------------------------------------------------

\medskip
\noindent\textbf{Proof of (i): Anthropomorphism increases attributed expectations.}

The human trustee's attributed second-order belief is $\tilde{h}_H^{(2,A)} = \phi_H(\rho_A, x_A, \omega_H)$. By (A2'), for $\omega_H' > \omega_H$:
\begin{equation}
    \phi_H(\rho_A, x_A, \omega_H') \geq \phi_H(\rho_A, x_A, \omega_H).
\end{equation}
Hence $\partial \tilde{h}_H^{(2,A)} / \partial \omega_H \geq 0$. \qed

%------------------------------------------------------------------------------
% PART (ii)
%------------------------------------------------------------------------------

\medskip
\noindent\textbf{Proof of (ii): Higher attributed expectations increase equilibrium returns.}

The human's utility is $U_H(y) = 3x - y - G \cdot \max\{0, \tilde{h} - y\}$, where $\tilde{h} = \tilde{h}_H^{(2,A)}$.

\emph{Step 1 (Piecewise structure).}
\begin{equation}
    U_H(y) = \begin{cases}
        3x - G\tilde{h} + (G-1)y & \text{if } y < \tilde{h}, \\
        3x - y & \text{if } y \geq \tilde{h}.
    \end{cases}
\end{equation}

\emph{Step 2 (Marginal utility).}
\begin{equation}
    \frac{\partial U_H}{\partial y} = \begin{cases}
        G - 1 > 0 & \text{if } y < \tilde{h} \text{ (by (G))}, \\
        -1 < 0 & \text{if } y > \tilde{h}.
    \end{cases}
\end{equation}

\emph{Step 3 (Optimal return).} The unique maximum is $y^* = \min\{\tilde{h}, 3x\}$.

\emph{Step 4 (Comparative statics).} $\partial y^* / \partial \tilde{h} = 1$ if $\tilde{h} < 3x$, and 0 otherwise. \qed

%------------------------------------------------------------------------------
% PART (iii)
%------------------------------------------------------------------------------

\medskip
\noindent\textbf{Proof of (iii): Same material payoffs, different equilibria.}

By Claims (i) and (ii), $y^*$ depends on $\omega_H$ through attributed beliefs. Consider $E = 10$, $\rho_A = 0.3$, $G = 1.2$, $x = 10$, with $\phi_H(\rho_A, x_A, \omega_H) = \omega_H \cdot \rho_A \cdot 3x$:
\begin{itemize}
    \item Human $H$ ($\omega_H = 0.8$): $\tilde{h}_H^{(2,A)} = 7.2$, $y^*_H = 7.2$.
    \item Human $H'$ ($\omega_H' = 0.3$): $\tilde{h}_{H'}^{(2,A)} = 2.7$, $y^*_{H'} = 2.7$.
\end{itemize}

Material payoff functions are identical ($30 - y$), yet equilibrium outcomes differ: anthropomorphism affects attributed beliefs, which determine guilt-driven returns. \qed
\end{proof}

\end{remark}
